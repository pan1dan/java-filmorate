Table films {
  film_id bigint [primary key]
  name varchar(100)
  description varchar(1000)
  release_date date
  duration integer
  mpa_id integer [ref : > film_rating_mpa.mpa_id] // one to one
}
Ref: films.film_id < film_genre.film_id // one to many
Ref : films.film_id <> film_director.film_id // many to many

Table film_director {
  film_id bigint
  director_id bigint [ref : - directors.director_id]
}

Table directors {
  director_id bigint [primary key]
  director_name varchar(100)
}

Table film_genre {
  film_id bigint
  genre_id integer [ref: - genres.genre_id] // one to one
}

Table genres {
  genre_id integer [primary key]
  genre_name varchar
}

Table film_rating_mpa {
  mpa_id integer [primary key]
  mpa_name varchar
}

Table users {
  user_id bigint [primary key]
  email varchar(100)
  login varchar(100)
  name varchar(100)
  birthday date
}

Enum friendship_status {
  UNCONFIRMED
  CONFIRMED
}

Table user_friends {
  user_id bigint [primary key]
  status friendship_status
  friend_id bigint

}

Table user_events {
  event_id BIGINT [primary key]
  user_id BIGINT
  event_type VARCHAR('LIKE', 'REVIEW', 'FRIEND')
  operation VARCHAR('REMOVE', 'ADD', 'UPDATE')
  entity_id BIGINT
  timestamp DATE
}

Ref: users.user_id < user_friends.user_id // one to many
Ref: user_friends.friend_id - users.user_id // one to one

Ref: users.user_id < user_events.user_id // one to many

Table users_likes_films {
  film_id bigint
  user_id bigint
}
Ref: users_likes_films.film_id > films.film_id
Ref: users_likes_films.user_id > users.user_id

Table usabilitys {
  usability_id int [primary key]
  usability varchar unique
  weigh int
}

Table reviews {
  review_id bigint [primary key]
  film_id bigint
  user_id bigint
  content varchar(500)
  is_positive bool
}

Table usability_reviews {
  usability_review_id bigint [primary key]
  user_id bigint
  review_id bigint
  usability_id int
}

Ref: users.user_id < reviews.user_id
Ref: films.film_id < reviews.film_id
Ref: users.user_id < usability_reviews.user_id
Ref: reviews.review_id < usability_reviews.review_id
Ref: usabilitys.usability_id - usability_reviews.usability_id
