Table film {
  film_id bigint [primary key]
  name varchar(100)
  description varchar(1000)
  release_date date
  duration integer
  mpa_id integer [ref : > film_rating_mpa.mpa_id] // one to one
}
Ref: film.film_id < film_genre.film_id // one to many
Ref : film.film_id <> film_director.film_id // many to many

Table film_director {
  film_id bigint
  director_id bigint [ref : - directors.director_id]
}

Table directors {
  director_id bigint [primary key]
  director_name varchar(100)
}

Table film_genre {
  film_id bigint
  genre_id integer [ref: - genres.genre_id] // one to one
}

Table genres {
  genre_id integer [primary key]
  genre_name varchar
}

Table film_rating_mpa {
  mpa_id integer [primary key]
  mpa_name varchar
}

Table user {
  user_id bigint [primary key]
  email varchar(100)
  login varchar(100)
  name varchar(100)
  birthday date
}

Enum friendship_status {
  UNCONFIRMED
  CONFIRMED
}

Table user_friend {
  user_id bigint [primary key]
  status friendship_status
  friend_id bigint

}

Table user_events {
  event_id BIGINT [primary key]
  user_id BIGINT
  event_type VARCHAR('LIKE', 'REVIEW', 'FRIEND')
  operation VARCHAR('REMOVE', 'ADD', 'UPDATE')
  entity_id BIGINT
  timestamp DATE
}

Ref: user.user_id < user_friend.user_id // one to many
Ref: user_friend.friend_id - user.user_id // one to one
Ref: user.user_id < user_events.user_id // one to many

Table user_like_film {
  film_id bigint
  user_id bigint
}
Ref: user_like_film.film_id > film.film_id
Ref: user_like_film.user_id > user.user_id
